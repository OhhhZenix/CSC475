start  App()

        string nav = "home" 
    
    if nav = "home" then
        HomeScreen(
            { nav = "storage" },
            { nav = "internet" }
        )
    endIf

    if nav = "storage" then
        StorageScreen(
            { nav = "home" }
        )
    endIf

    if nav = "internet" then
        StorageScreen(
            { nav = "home" }
        )
    endIf
end App
