start  App()
    Declarations
        string nav = "home" 
    
    if nav = "home" then
        HomeScreen(
            { nav = "storage" },
            { nav = "internet" }
        )
    endIf

    if nav = "storage" then
        StorageScreen(
            { nav = "home" }
        )
    endIf

    if nav = "internet" then
        StorageScreen(
            { nav = "home" }
        )
    endIf
end App

start HomeScreen(navStorageScreen, navInternetScreen)
    start Scaffold
        set top bar with title "Photo Gallery"

        stat Box
            set content alignment to center
            set fill max size

            start Column
                set padding to 8
                set align to top center
                set offset to 25% of screen
                set vertical arrangement to center
                set horizontal alignment to center

                stat Text
                    set content to "Welcome!"
                    set font size to 30
                    set text align to center
                end Text
                
                start Text
                    set content to "To get started, select the options below."
                    set text align to center
                end Text

                addSpacer(8)

                start Button
                    set on click to navStorageScreen
                    set fill max width

                    start Text
                        set content to "View from storage"
                    end Text
                end Button

                start Button
                    set on click to navInternetScreen
                    set fill max width

                    start Text
                        set content to "View from internet"
                    end Text
                end Button
            end Column
        end Box
    end Scaffold
end HomeScreen

start StorageScreen(navHomeScreen)
    start Scaffold
        set top bar with title "Photo Gallery"

        start Column
            set fill max size

            start Button
                set on click to navHomeScreen
                set padding to 8

                start Text
                    set content to "Get Started"
                    set fill max width
                    set text align to center
                    set font size to 32
                end Text

                start Button
                    on click to launch media selector with 10 limit
                    set fill max width
                    set padding to 8

                    start Text
                        set content to "Pick image from storage"
                    end Text
                end Button

                if images is not empty then
                    start Text
                      set content to "No images are selected currently."
                      set fill max width
                      set text align to center 
                    end Text
                else
                    start Text
                        set content to "Selected images:"
                        set offset x to 8
                    end Text

                    start LazyColumn
                        set fill max size

                        for image in images
                            start AsyncImage
                                set model to image
                                set content description to empty
                                set fill max width
                                set height to 250
                                set padding to 8
                                set background to black
                                set content scale to fit
                            end AsyncImage
                        endFor
                    end LazyColumn
                endIf
            end Button
        end Column
    end Scaffold
end StorageScreen
